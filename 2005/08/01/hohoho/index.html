Tus gw iseng negh ceritanya, bikin shell script.

#!/bin/sh

TRUNK=~/trunk
WPPATH=blog
TARGZFILE=wp.tar.gz

sendftp () {
 if ./ftp.tcl ; then
   echo "SUKSES";
 else
   echo "GAGAL";
 fi
}

cd ~
if [ -d $TRUNK ] ; then
  cd $TRUNK
  if [ -d $TRUNK/.svn ] ; then
    if svn co http://svn.automattic.com/wordpress/trunk/ $TRUNK ; then
    else
      exit 3
    fi
  else
    svn update
  fi
  cd ~
else
  if svn co http://svn.automattic.com/wordpress/trunk/ $TRUNK ; then
  else
    exit 2
  fi
fi

rm -rf $WPPATH ; cp -R $TRUNK $WPPATH
cd $WPPATH ; rm -rf `find . | grep svn` ; cd ~
rm -rf $TARGZFILE ; tar -zcf $TARGZFILE $WPPATH ; rm -rf $WPPATH

sendftp

exit 0

Edit file ftp.tcl (sebenernya ini tcl script buat di eggdrop) di bagian ftpserver, username sama path nya. jangan lupa: chmod u+x ftp.tcl sama code di atas tadi.
